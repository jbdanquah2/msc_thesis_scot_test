# Input: s: string
# Output: integer (number of vowels in s)
# 1: set vowels to "aeiouAEIOU"
# 2: set count to 0
# 3: for each character c in s do
# 4:     if c is in vowels then
# 5:         increment count by 1
# 6: return count

def count_vowels(s: str) -> int:
    vowels = "aeiouAEIOU"
    count = 0
    for c in s:
        if c in vowels:
            count += 1
    return count
