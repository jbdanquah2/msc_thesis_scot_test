# Input: A: list of lists of integers
#        B: list of lists of integers
# Output: list of lists of integers (product of A and B)
# 1: set n to the number of rows in A
# 2: set m to the number of columns in A (also the number of rows in B)
# 3: set p to the number of columns in B
# 4: initialize result as an n x p matrix with all elements as 0
# 5: for i from 0 to n-1 do
# 6:     for j from 0 to p-1 do
# 7:         for k from 0 to m-1 do
# 8:             result[i][j] += A[i][k] * B[k][j]
# 9: return result
# Final Code
def matrix_multiply(A: list[list[int]], B: list[list[int]]) -> list[list[int]]:
    n = len(A)
    m = len(A[0])
    p = len(B[0])
    result = [[0 for _ in range(p)] for _ in range(n)]

    for i in range(n):
        for j in range(p):
            for k in range(m):
                result[i][j] += A[i][k] * B[k][j]

    return result
