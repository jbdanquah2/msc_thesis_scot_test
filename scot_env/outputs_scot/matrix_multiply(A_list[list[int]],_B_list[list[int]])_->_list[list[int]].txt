
    # Input: A, B: list of lists of integers
    # Output: list of lists of integers (product of A and B)
    # 1: set rows_A to number of rows in A
    # 2: set cols_A to number of columns in A
    # 3: set cols_B to number of columns in B
    # 4: initialize result as a matrix of size rows_A x cols_B filled with zeros
    # 5: for i from 0 to rows_A - 1 do
    # 6:     for j from 0 to cols_B - 1 do
    # 7:         for k from 0 to cols_A - 1 do
    # 8:             result[i][j] += A[i][k] * B[k][j]
    # 9: return result
    ### Final Code
def matrix_multiply(A: list[list[int]], B: list[list[int]]) -> list[list[int]]:
    rows_A, cols_A = len(A), len(A[0])
    cols_B = len(B[0])
    result = [[0 for _ in range(cols_B)] for _ in range(rows_A)]

    for i in range(rows_A):
        for j in range(cols_B):
            for k in range(cols_A):
                result[i][j] += A[i][k] * B[k][j]

    return result
